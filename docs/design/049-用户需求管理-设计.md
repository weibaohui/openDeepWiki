# 1. 设计概述

本设计文档描述用户需求管理功能的实现方案，包括数据模型设计、后端 API 设计、前端界面设计等。该功能旨在为用户提供便捷的需求反馈机制，并为管理员提供需求管理界面。

# 2. 数据模型设计

## 2.1 UserRequest 数据模型

创建新的数据模型 `UserRequest`，用于存储用户提交的需求。

### 模型定义

```go
// backend/internal/model/user_request.go

package model

import "time"

// UserRequest 用户需求模型
type UserRequest struct {
    ID           uint      `json:"id" gorm:"primaryKey"`
    RepositoryID uint      `json:"repository_id" gorm:"index;not null"`
    Content      string    `json:"content" gorm:"type:text;not null;size:200"`
    Status       string    `json:"status" gorm:"size:50;not null;default:pending"`
    CreatedAt    time.Time `json:"created_at"`
    UpdatedAt    time.Time `json:"updated_at"`

    // 关联
    Repository   *Repository `json:"repository,omitempty" gorm:"foreignKey:RepositoryID"`
}
```

### 字段说明

| 字段 | 类型 | 说明 | 约束 |
|------|------|------|------|
| ID | uint | 主键，自增 | PRIMARY KEY |
| RepositoryID | uint | 关联的仓库 ID | NOT NULL, INDEX |
| Content | string | 用户需求内容 | NOT NULL, TEXT, SIZE:200 |
| Status | string | 需求状态 | NOT NULL, DEFAULT: 'pending' |
| CreatedAt | time.Time | 创建时间 | 自动生成 |
| UpdatedAt | time.Time | 更新时间 | 自动更新 |

### 状态枚举

| 状态 | 说明 |
|------|------|
| pending | 待处理 |
| processing | 处理中 |
| completed | 已完成 |
| rejected | 已拒绝 |

```go
package repository

import (
    "gorm.io/gorm"
    "github.com/weibaohui/opendeepwiki/backend/internal/model"
)

// UserRequestRepository 用户需求仓库接口
type UserRequestRepository interface {
    Create(request *model.UserRequest) error
    GetByID(id uint) (*model.UserRequest, error)
    ListByRepository(repoID uint, page, pageSize int, status string) ([]*model.UserRequest, int64, error)
    Delete(id uint) error
    UpdateStatus(id uint, status string) error
}

// userRequestRepository 用户需求仓库实现
type userRequestRepository struct {
    db *gorm.DB
}

// NewUserRequestRepository 创建用户需求仓库
func NewUserRequestRepository(db *gorm.DB) UserRequestRepository {
    return &userRequestRepository{db: db}
}

func (r *userRequestRepository) Create(request *model.UserRequest) error {
    return r.db.Create(request).Error
}

func (r *userRequestRepository) GetByID(id uint) (*model.UserRequest, error) {
    var request model.UserRequest
    err := r.db.Preload("Repository").First(&request, id).Error
    return &request, err
}

func (r *userRequestRepository) ListByRepository(repoID uint, page, pageSize int, status string) ([]*model.UserRequest, int64, error) {
    var requests []*model.UserRequest
    var total int64

    query := r.db.Model(&model.UserRequest{}).Where("repository_id = ?", repoID)

    if status != "" {
        query = query.Where("status = ?", status)
    }

    // 获取总数
    query.Count(&total)

    // 分页查询
    offset := (page - 1) * pageSize
    err := query.Order("created_at DESC").Offset(offset).Limit(pageSize).Find(&requests).Error

    return requests, total, err
}

func (r *userRequestRepository) Delete(id uint) error {
    return r.db.Delete(&model.UserRequest{}, id).Error
}

func (r *userRequestRepository) UpdateStatus(id uint, status string) error {
    return r.db.Model(&model.UserRequest{}).Where("id = ?", id).Update("status", status).Error
}
```

## 3.2 Service 层

### 文件：backend/internal/service/user_request.go

```go
package service

import (
    "errors"
    "fmt"
    "github.com/weibaohui/opendeepwiki/backend/internal/model"
    "github.com/weibaohui/opendeepwiki/backend/internal/repository"
    "k8s.io/klog/v2"
)

// UserRequestService 用户需求服务接口
type UserRequestService interface {
    CreateRequest(repoID uint, content string) (*model.UserRequest, error)
    GetRequest(id uint) (*model.UserRequest, error)
    ListRequests(repoID uint, page, pageSize int, status string) ([]*model.UserRequest, int64, error)
    DeleteRequest(id uint) error
    UpdateStatus(id uint, status string) error
}

// userRequestService 用户需求服务实现
type userRequestService struct {
    userRequestRepo repository.UserRequestRepository
    repoRepo        repository.Repository
}

// NewUserRequestService 创建用户需求服务
func NewUserRequestService(userRequestRepo repository.UserRequestRepository, repoRepo repository.Repository) UserRequestService {
    return &userRequestService{
        userRequestRepo: userRequestRepo,
        repoRepo:        repoRepo,
    }
}

func (s *userRequestService) CreateRequest(repoID uint, content string) (*model.UserRequest, error) {
    klog.V(6).Infof("[UserRequestService] 创建用户需求: repoID=%d, content=%s", repoID, content)

    // 验证内容
    if content == "" {
        return nil, errors.New("需求内容不能为空")
    }

    if len(content) > 200 {
        return nil, errors.New("需求内容不能超过200个字符")
    }

    // 验证仓库是否存在
    _, err := s.repoRepo.GetBasic(repoID)
    if err != nil {
        return nil, fmt.Errorf("仓库不存在: %w", err)
    }

    // 创建用户需求
    request := &model.UserRequest{
        RepositoryID: repoID,
        Content:      content,
        Status:       "pending",
    }

    if err := s.userRequestRepo.Create(request); err != nil {
        klog.Errorf("[UserRequestService] 创建用户需求失败: %v", err)
        return nil, fmt.Errorf("保存需求失败: %w", err)
    }

    klog.V(6).Infof("[UserRequestService] 用户需求创建成功: id=%d", request.ID)
    return request, nil
}

func (s *userRequestService) GetRequest(id uint) (*model.UserRequest, error) {
    request, err := s.userRequestRepo.GetByID(id)
    if err != nil {
        klog.Errorf("[UserRequestService] 获取用户需求失败: id=%d, error=%v", id, err)
        return nil, fmt.Errorf("获取需求失败: %w", err)
    }
    return request, nil
}

func (s *userRequestService) ListRequests(repoID uint, page, pageSize int, status string) ([]*model.UserRequest, int64, error) {
    klog.V(6).Infof("[UserRequestService] 获取用户需求列表: repoID=%d, page=%d, pageSize=%d, status=%s", repoID, page, pageSize, status)

    requests, total, err := s.userRequestRepo.ListByRepository(repoID, page, pageSize, status)
    if err != nil {
        klog.Errorf("[UserRequestService] 获取用户需求列表失败: %v", err)
        return nil, 0, fmt.Errorf("获取需求列表失败: %w", err)
    }

    return requests, total, nil
}

func (s *userRequestService) DeleteRequest(id uint) error {
    klog.V(6).Infof("[UserRequestService] 删除用户需求: id=%d", id)

    if err := s.userRequestRepo.Delete(id); err != nil {
        klog.Errorf("[UserRequestService] 删除用户需求失败: %v", err)
        return fmt.Errorf("删除需求失败: %w", err)
    }

    return nil
}

func (s *userRequestService) UpdateStatus(id uint, status string) error {
    klog.V(6).Infof("[UserRequestService] 更新用户需求状态: id=%d, status=%s", id, status)

    if err := s.userRequestRepo.UpdateStatus(id, status); err != nil {
        klog.Errorf("[UserRequestService] 更新用户需求状态失败: %v", err)
        return fmt.Errorf("更新需求状态失败: %w", err)
    }

    return nil
}
```

## 3.3 Handler 层

```go
package handler

import (
    "net/http"
    "strconv"

    "github.com/gin-gonic/gin"
    "github.com/weibaohui/opendeepwiki/backend/internal/service"
)

// UserRequestHandler 用户需求处理器
type UserRequestHandler struct {
    service service.UserRequestService
}

// NewUserRequestHandler 创建用户需求处理器
func NewUserRequestHandler(service service.UserRequestService) *UserRequestHandler {
    return &UserRequestHandler{service: service}
}

// CreateUserRequest 创建用户需求
func (h *UserRequestHandler) CreateUserRequest(c *gin.Context) {
    repoID, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"code": 400, "message": "invalid repository id"})
        return
    }

    var req struct {
        Content string `json:"content" binding:"required"`
    }

    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"code": 400, "message": "content is required"})
        return
    }

    request, err := h.service.CreateRequest(uint(repoID), req.Content)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"code": 500, "message": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "code":    0,
        "message": "需求已提交",
        "data":    request,
    })
}

// ListUserRequests 获取用户需求列表
func (h *UserRequestHandler) ListUserRequests(c *gin.Context) {
    repoID, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"code": 400, "message": "invalid repository id"})
        return
    }

    page := 1
    pageSize := 20
    status := c.Query("status")

    if pageStr := c.Query("page"); pageStr != "" {
        if p, err := strconv.Atoi(pageStr); err == nil && p > 0 {
            page = p
        }
    }

    if sizeStr := c.Query("page_size"); sizeStr != "" {
        if s, err := strconv.Atoi(sizeStr); err == nil && s > 0 && s <= 50 {
            pageSize = s
        }
    }

    requests, total, err := h.service.ListRequests(uint(repoID), page, pageSize, status)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"code": 500, "message": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "code":     0,
        "message":  "success",
        "data": gin.H{
            "list":      requests,
            "total":     total,
            "page":      page,
            "page_size": pageSize,
        },
    })
}

// GetUserRequest 获取用户需求详情
func (h *UserRequestHandler) GetUserRequest(c *gin.Context) {
    id, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"code": 400, "message": "invalid id"})
        return
    }

    request, err := h.service.GetRequest(uint(id))
    if err != nil {
        c.JSON(http.StatusNotFound, gin.H{"code": 404, "message": "request not found"})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "code":    0,
        "message": "success",
        "data":    request,
    })
}

// DeleteUserRequest 删除用户需求
func (h *UserRequestHandler) DeleteUserRequest(c *gin.Context) {
    id, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"code": 400, "message": "invalid id"})
        return
    }

    if err := h.service.DeleteRequest(uint(id)); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"code": 500, "message": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "code":    0,
        "message": "删除成功",
        "data":    nil,
    })
}
```

## 4.3 用户需求列表页面

### 文件：frontend/src/pages/UserRequestList.tsx

```typescript
import { useState, useEffect } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { ArrowLeftOutlined, DeleteOutlined, ReloadOutlined } from '@ant-design/icons';
import { Button, Card, Spin, Layout, Typography, Table, Space, Modal, Tag, Select, Empty } from 'antd';
import type { UserRequest } from '../types';
import { userRequestApi } from '../services/api';
import { useAppConfig } from '@/context/AppConfigContext';

const { Header, Content } = Layout;
const { Title } = Typography;

export default function UserRequestList() {
    const { t } = useAppConfig();
    const { id } = useParams<{ id: string }>();
    const navigate = useNavigate();
    const [requests, setRequests] = useState<UserRequest[]>([]);
    const [loading, setLoading] = useState(true);
    const [total, setTotal] = useState(0);
    const [page, setPage] = useState(1);
    const [pageSize, setPageSize] = useState(20);
    const [statusFilter, setStatusFilter] = useState<string | undefined>(undefined);
    const [messageApi, contextHolder] = message.useMessage();

    const fetchRequests = async () => {
        if (!id) return;
        setLoading(true);
        try {
            const { data } = await userRequestApi.list(Number(id), {
                page,
                page_size: pageSize,
                status: statusFilter,
            });
            setRequests(data.list || []);
            setTotal(data.total || 0);
        } catch (error) {
            console.error('Failed to fetch requests:', error);
            messageApi.error('Failed to load requests');
        } finally {
            setLoading(false);
        }
    };

    useEffect(() => {
        fetchRequests();
    }, [id, page, pageSize, statusFilter]);

    const handleDelete = (id: number) => {
        Modal.confirm({
            title: 'Confirm Delete',
            content: 'Are you sure you want to delete this request?',
            okText: 'Delete',
            okType: 'danger',
            cancelText: 'Cancel',
            onOk: async () => {
                try {
                    await userRequestApi.delete(id);
                    messageApi.success('Request deleted');
                    fetchRequests();
                } catch (error) {
                    console.error('Failed to delete request:', error);
                    messageApi.error('Failed to delete request');
                }
            },
        });
    };

    const getStatusColor = (status: string) => {
        switch (status) {
            case 'pending': return 'default';
            case 'processing': return 'processing';
            case 'completed': return 'success';
            case 'rejected': return 'error';
            default: return 'default';
        }
    };

    const columns = [
        {
            title: 'Content',
            dataIndex: 'content',
            key: 'content',
            ellipsis: true,
        },
        {
            title: 'Status',
            dataIndex: 'status',
            key: 'status',
            width: 120,
            render: (status: string) => <Tag color={getStatusColor(status)}>{status}</Tag>,
        },
        {
            title: 'Created At',
            dataIndex: 'created_at',
            key: 'created_at',
            width: 180,
            render: (date: string) => new Date(date).toLocaleString(),
        },
        {
            title: 'Action',
            key: 'action',
            width: 100,
            render: (_, record: UserRequest) => (
                <Button
                    type="text"
                    danger
                    icon={<DeleteOutlined />}
                    onClick={() => handleDelete(record.id)}
                />
            ),
        },
    ];

    if (loading) {
        return (
            <div style={{ display: 'flex', justifyContent: 'center', alignItems: 'center', height: '100vh' }}>
                <Spin size="large" />
            </div>
        );
    }

    return (
        <Layout style={{ minHeight: '100vh' }}>
            {contextHolder}
            <Header style={{
                display: 'flex',
                alignItems: 'center',
                padding: '0 24px',
                background: 'var(--ant-color-bg-container)',
                borderBottom: '1px solid var(--ant-color-border-secondary)'
            }}>
                <Button
                    type="text"
                    icon={<ArrowLeftOutlined />}
                    onClick={() => navigate(`/repo/${id}`)}
                    style={{ marginRight: 8 }}
                />
                <Title level={4} style={{ margin: 0 }}>User Requests</Title>
            </Header>
            <Content style={{ padding: '24px', maxWidth: '1200px', margin: '0 auto', width: '100%' }}>
                <Card
                    title="User Requests"
                    extra={
                        <Space>
                            <Select
                                placeholder="Filter by status"
                                style={{ width: 150 }}
                                allowClear
                                onChange={(value) => setStatusFilter(value || undefined)}
                            >
                                <Select.Option value="pending">Pending</Select.Option>
                                <Select.Option value="processing">Processing</Select.Option>
                                <Select.Option value="completed">Completed</Select.Option>
                                <Select.Option value="rejected">Rejected</Select.Option>
                            </Select>
                            <Button
                                icon={<ReloadOutlined />}
                                onClick={fetchRequests}
                            >
                                Refresh
                            </Button>
                        </Space>
                    }
                >
                    {requests.length === 0 ? (
                        <Empty description="No user requests found" />
                    ) : (
                        <Table
                            dataSource={requests}
                            columns={columns}
                            rowKey="id"
                            pagination={{
                                current: page,
                                pageSize: pageSize,
                                total: total,
                                onChange: (p, ps) => {
                                    setPage(p);
                                    setPageSize(ps || 20);
                                },
                            }}
                        />
                    )}
                </Card>
            </Content>
        </Layout>
    );
}
```

# 5. 安全考虑

1. **输入验证**：前端和后端都需要验证用户输入，确保内容非空且长度不超过 200 字符
2. **权限控制**：只有仓库管理员才能删除用户需求
3. **防 SQL 注入**：使用 GORM 参数化查询，避免直接拼接 SQL
4. **防 XSS**：前端使用 React 自动转义，显示用户输入时进行额外的转义处理
5. **日志记录**：记录所有用户需求的创建、删除、状态变更操作

# 6. 性能考虑

1. **分页查询**：列表查询使用分页，避免一次性加载大量数据
2. **索引优化**：为 repository_id 和 status 字段创建索引，提高查询效率
3. **缓存策略**：可以考虑对频繁访问的数据进行缓存
4. **异步处理**：创建用户需求时使用异步处理，避免阻塞主线程
