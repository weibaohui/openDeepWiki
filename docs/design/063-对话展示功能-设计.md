## 1. 设计目标

### 1.1 要解决的问题

1. **对话无法被共享**：当前 DocCopilot 的对话仅对参与者可见，知识无法沉淀和分享
2. **缺少对话管理入口**：用户无法像管理文档一样浏览和管理历史对话
3. **对话内容无法复用**：有价值的问答无法被其他人学习和参考

### 1.2 设计原则

1. **一致性**：对话展示的体验与文档展示保持一致
2. **安全性**：私有对话严格保护，仅创建者可访问
3. **复用性**：最大化复用现有组件和逻辑

    repoID, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "无效的仓库ID"})

    pageSize, _ := strconv.Atoi(c.DefaultQuery("page_size", "20"))

    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "items":     sessions,
        "total":     total,
        "page":      page,
        "page_size": pageSize,
    })
}

// GetSessionView 获取对话详情（供展示使用）
func (h *ChatHandler) GetSessionView(c *gin.Context) {
    repoID, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "无效的仓库ID"})
        return
    }

    sessionID := c.Param("session_id")

    // 获取会话
    session, err := h.chatService.GetSession(c.Request.Context(), sessionID)
    if err != nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "会话不存在"})
        return
    }

    // 验证会话属于该仓库
    if session.RepoID != uint(repoID) {
        c.JSON(http.StatusForbidden, gin.H{"error": "会话不属于该仓库"})
        return
    }

    // 权限检查
    // TODO: 获取当前用户ID
    currentUserID := uint(0) // 临时
    if session.Visibility == "private" && session.CreatedBy != currentUserID {
        c.JSON(http.StatusForbidden, gin.H{"error": "无权限查看此对话"})
        return
    }

    // 获取消息列表
    messages, err := h.chatService.ListMessages(c.Request.Context(), sessionID, 100, nil)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "session":  session,
        "messages": messages,
    })
}

func (h *ChatHandler) UpdateSessionVisibility(c *gin.Context) {
    repoID, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "无效的仓库ID"})
        return
    }

    sessionID := c.Param("session_id")

    var req struct {
        Visibility string `json:"visibility" binding:"required,oneof=public private"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "无效的请求参数"})
        return
    }

    // 获取会话
    session, err := h.chatService.GetSession(c.Request.Context(), sessionID)
    if err != nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "会话不存在"})
        return
    }

    // 验证会话属于该仓库
    if session.RepoID != uint(repoID) {
        c.JSON(http.StatusForbidden, gin.H{"error": "会话不属于该仓库"})
        return
    }

    // 权限检查：仅创建者可修改
    // TODO: 获取当前用户ID
    currentUserID := uint(0) // 临时
    if session.CreatedBy != currentUserID {
        c.JSON(http.StatusForbidden, gin.H{"error": "无权限修改此对话"})
        return
    }

    // 更新可见性
    if err := h.chatService.UpdateSessionVisibility(c.Request.Context(), sessionID, req.Visibility); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "session_id": sessionID,
        "visibility": req.Visibility,
    })
}

// backend/internal/service/chat_service.go

// ListPublicSessions 获取公开对话列表
func (s *chatService) ListPublicSessions(ctx context.Context, repoID uint, page, pageSize int) ([]model.ChatSession, int64, error) {
    var sessions []model.ChatSession
    var total int64

    offset := (page - 1) * pageSize

    err := s.db.WithContext(ctx).
        Model(&model.ChatSession{}).
        Where("repository_id = ? AND visibility = ?", repoID, "public").
        Count(&total).
        Order("updated_at DESC").
        Offset(offset).
        Limit(pageSize).
        Find(&sessions).Error

    return sessions, total, err
}

// UpdateSessionVisibility 更新对话可见性
func (s *chatService) UpdateSessionVisibility(ctx context.Context, sessionID, visibility string) error {
    return s.db.WithContext(ctx).
        Model(&model.ChatSession{}).
        Where("session_id = ?", sessionID).
        Update("visibility", visibility).Error
}

// UpdateMessageCount 更新消息数量（在消息变化时调用）
func (s *chatService) UpdateMessageCount(ctx context.Context, sessionID string) error {
    var count int64
    s.db.WithContext(ctx).
        Model(&model.ChatMessage{}).
        Where("session_id = ?", sessionID).
        Count(&count)

    return s.db.WithContext(ctx).
        Model(&model.ChatSession{}).
        Where("session_id = ?", sessionID).
        Update("message_count", count).Error
}

// frontend/src/components/chat/ChatSessionViewer.tsx

import { useEffect, useState } from 'react';
import { Card, Spin, Typography, Collapse, Tag } from 'antd';
import { RobotOutlined, UserOutlined, ToolOutlined } from '@ant-design/icons';
import type { ChatSession, ChatMessage } from '@/types/chat';
import { chatApi } from '@/services/api';

interface ChatSessionViewerProps {
  repoId: number;
  sessionId: string;
}

export function ChatSessionViewer({ repoId, sessionId }: ChatSessionViewerProps) {
  const [loading, setLoading] = useState(true);
  const [session, setSession] = useState<ChatSession | null>(null);
  const [messages, setMessages] = useState<ChatMessage[]>([]);

  useEffect(() => {
    const fetchSession = async () => {
      try {
        const { data } = await chatApi.getSessionView(repoId, sessionId);
        setSession(data.session);
        setMessages(data.messages);
      } catch (error) {
        console.error('Failed to fetch session:', error);
      } finally {
        setLoading(false);
      }
    };
    fetchSession();
  }, [repoId, sessionId]);

  if (loading) return <Spin size="large" />;
  if (!session) return <Typography.Text>对话不存在或无权限查看</Typography.Text>;

  return (
    <div style={{ maxWidth: 900, margin: '0 auto' }}>
      <Typography.Title level={3}>{session.title}</Typography.Title>

      {messages.map((msg) => (
        <ChatMessageView key={msg.message_id} message={msg} />
      ))}
    </div>
  );
}

// 单条消息展示
function ChatMessageView({ message }: { message: ChatMessage }) {
  const isUser = message.role === 'user';

  return (
    <Card
      style={{
        marginBottom: 16,
        background: isUser ? 'var(--ant-color-primary-bg)' : 'var(--ant-color-bg-container)',
      }}
    >
      <div style={{ display: 'flex', gap: 12 }}>
        <div>
          {isUser ? (
            <UserOutlined style={{ fontSize: 24 }} />
          ) : (
            <RobotOutlined style={{ fontSize: 24, color: '#10a37f' }} />
          )}
        </div>
        <div style={{ flex: 1 }}>
          <Typography.Text strong>
            {isUser ? '用户' : 'AI'}
          </Typography.Text>

          {/* 消息内容 */}
          <div style={{ marginTop: 8 }}>
            <Typography.Text>{message.content}</Typography.Text>
          </div>

          {/* 思考过程（AI消息） */}
          {!isUser && message.thinking_content && (
            <Collapse ghost style={{ marginTop: 8 }}>
              <Collapse.Panel header="思考过程" key="thinking">
                <Typography.Text type="secondary">
                  {message.thinking_content}
                </Typography.Text>
              </Collapse.Panel>
            </Collapse>
          )}

          {/* 工具调用（AI消息） */}
          {!isUser && message.tool_calls && message.tool_calls.length > 0 && (

              <Collapse.Panel
                header={<span><ToolOutlined /> 工具调用 ({message.tool_calls.length})</span>}

              >
                {message.tool_calls.map((tool) => (
                  <Card key={tool.tool_call_id} size="small" style={{ marginBottom: 8 }}>

// frontend/src/components/chat/ChatVisibilityToggle.tsx

import { useState } from 'react';
import { Switch, message, Tooltip } from 'antd';
import { GlobalOutlined, LockOutlined } from '@ant-design/icons';
import { chatApi } from '@/services/api';

interface ChatVisibilityToggleProps {
  repoId: number;
  sessionId: string;
  initialVisibility: 'public' | 'private';
  onChange?: (visibility: 'public' | 'private') => void;
}

export function ChatVisibilityToggle({
  repoId,
  sessionId,
  initialVisibility,
  onChange,
}: ChatVisibilityToggleProps) {
  const [visibility, setVisibility] = useState(initialVisibility);
  const [loading, setLoading] = useState(false);

  const handleChange = async (checked: boolean) => {
    const newVisibility = checked ? 'public' : 'private';
    setLoading(true);

    try {
      await chatApi.updateVisibility(repoId, sessionId, newVisibility);
      setVisibility(newVisibility);
      onChange?.(newVisibility);
      message.success(newVisibility === 'public' ? '已设为公开' : '已设为私有');
    } catch (error) {
      message.error('设置失败');
    } finally {
      setLoading(false);
    }
  };

  return (
    <Tooltip title={visibility === 'public' ? '公开：所有用户可见' : '私有：仅自己可见'}>
      <Switch
        checked={visibility === 'public'}
        onChange={handleChange}
        loading={loading}
        checkedChildren={<GlobalOutlined />}
        unCheckedChildren={<LockOutlined />}
      />
    </Tooltip>
  );
}

## 7. 即时响应实现策略

### 7.1 乐观更新（Optimistic Update）

为了提供即时的用户反馈，采用乐观更新策略：

#### 7.1.1 useChat Hook 扩展

```typescript
// frontend/src/hooks/useChat.ts

// 新增：更新本地会话状态
const updateLocalSession = useCallback((sessionId: string, updates: Partial<ChatSession>) => {
  setState((prev) => {
    const updatedSessions = prev.sessions.map((s) =>
      s.session_id === sessionId ? { ...s, ...updates } : s
    );
    const updatedCurrentSession =
      prev.currentSession?.session_id === sessionId
        ? { ...prev.currentSession, ...updates }
        : prev.currentSession;
    return {
      ...prev,
      sessions: updatedSessions,
      currentSession: updatedCurrentSession,
    };
  });
}, []);
```

#### 7.1.2 DocCopilot 中的使用

```typescript
// 公开/私有切换处理
onClick={async () => {
  const sessionId = state.currentSession!.session_id;
  const newVisibility = state.currentSession?.visibility === 'public' ? 'private' : 'public';

  // 1. 立即更新本地状态（乐观更新）
  updateLocalSession(sessionId, { visibility: newVisibility });

  try {
    // 2. 调用 API
    await chatApi.updateVisibility(repoId, sessionId, newVisibility);
    message.success(newVisibility === 'public' ? '已设为公开' : '已取消公开');
  } catch {
    // 3. 失败时回滚状态
    updateLocalSession(sessionId, { visibility: state.currentSession?.visibility });
    message.error('设置失败');
  }
}}
```

### 7.2 UI 状态对应

| 当前状态 | 显示按钮 | 点击后状态 | 操作 |
|---------|---------|-----------|------|
| private | 蓝色「设为公开」 | public | 立即变绿色「已公开（点击取消）」 |
| public | 绿色「已公开（点击取消）」 | private | 立即变蓝色「设为公开」 |
