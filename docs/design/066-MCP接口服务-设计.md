## 1. 设计概述

本文档描述 MCP (Model Context Protocol) 接口服务的技术设计方案，为 AI 编程工具提供文档查询能力。

### 1.1 设计目标

- 实现 MCP 协议服务端，支持 SSE 传输
- 复用现有 Service 层，最小化代码改动
- 保持与主服务的内聚，无需独立部署

### 1.2 技术选型

| 组件 | 技术选型 | 说明 |
|------|----------|------|
| MCP 框架 | [mcp-go](https://github.com/mark3labs/mcp-go) | 开源 Go MCP 实现 |
| HTTP 框架 | Gin | 复用现有框架，集成 mcp-go SSE |
| MCP 协议版本 | 2024-11-05 | mcp-go 向后兼容 |
| 数据访问 | 复用 Service | RepositoryService, DocumentService |

### 1.3 mcp-go 库优势

- 完整实现 MCP 协议规范
- 支持 stdio、SSE、streamable-HTTP 传输
- 高层次 API，代码简洁
- 自动处理协议细节（握手、消息格式等）

## 4. 核心组件设计

### 4.1 MCP Server (server.go)

使用 mcp-go 库创建 MCP Server 并注册工具：

```go
package mcp

import (
    "github.com/mark3labs/mcp-go/mcp"
    "github.com/mark3labs/mcp-go/server"

    "openDeepWiki/internal/service"
)

type MCPServerWrapper struct {
    server       *server.MCPServer
    repoService  *service.RepositoryService
    docService   *service.DocumentService
}

func NewMCPServer(repoSvc *service.RepositoryService, docSvc *service.DocumentService) *MCPServerWrapper {
    // 创建 mcp-go Server
    s := server.NewMCPServer(
        "openDeepWiki",
        "1.0.0",
        server.WithToolCapabilities(false),
        server.WithRecovery(),
    )

    wrapper := &MCPServerWrapper{
        server:      s,
        repoService: repoSvc,
        docService:  docSvc,
    }

    // 注册工具
    wrapper.registerTools()

    return wrapper
}

func (w *MCPServerWrapper) GetServer() *server.MCPServer {
    return w.server
}

func (w *MCPServerWrapper) registerTools() {
    // 1. list_repositories
    w.server.AddTool(mcp.NewTool("list_repositories",
        mcp.WithDescription("列出所有可用的代码仓库"),
    ), w.handleListRepositories)

    // 2. get_repository
    w.server.AddTool(mcp.NewTool("get_repository",
        mcp.WithDescription("获取仓库详情，包含文档列表"),
        mcp.WithNumber("repo_id",
            mcp.Description("仓库ID"),
        ),
        mcp.WithString("repo_name",
            mcp.Description("仓库名称（与 repo_id 二选一）"),
        ),
    ), w.handleGetRepository)

    // 3. search_documents
    w.server.AddTool(mcp.NewTool("search_documents",
        mcp.WithDescription("搜索文档内容"),
        mcp.WithString("query",
            mcp.Required(),
            mcp.Description("搜索关键词"),
        ),
        mcp.WithNumber("repo_id",
            mcp.Description("限定仓库ID（可选）"),
        ),
    ), w.handleSearchDocuments)

    // 4. read_document
    w.server.AddTool(mcp.NewTool("read_document",
        mcp.WithDescription("读取文档完整内容"),
        mcp.WithNumber("doc_id",
            mcp.Required(),
            mcp.Description("文档ID"),
        ),
    ), w.handleReadDocument)
}
```

### 4.2 Tool Handlers (handlers.go)

实现具体的工具处理器：

```go
package mcp

import (
    "context"
    "encoding/json"
    "fmt"

    "github.com/mark3labs/mcp-go/mcp"
    "k8s.io/klog/v2"
)

// handleListRepositories 列出所有仓库
func (w *MCPServerWrapper) handleListRepositories(ctx context.Context, request mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    klog.V(6).Info("MCP: handleListRepositories called")

    repos, err := w.repoService.GetAllRepositories(ctx)
    if err != nil {
        klog.Errorf("MCP: list repositories failed: %v", err)
        return mcp.NewToolResultError(fmt.Sprintf("获取仓库列表失败: %v", err)), nil
    }

    // 构建返回结果
    result := map[string]interface{}{
        "repositories": repos,
    }

    data, _ := json.MarshalIndent(result, "", "  ")
    return mcp.NewToolResultText(string(data)), nil
}

// handleGetRepository 获取仓库详情
func (w *MCPServerWrapper) handleGetRepository(ctx context.Context, request mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    klog.V(6).Info("MCP: handleGetRepository called")

    // 解析参数
    repoID, _ := request.RequireInt("repo_id")
    repoName := request.GetString("repo_name", "")

    var repo *model.Repository
    var err error

    if repoID > 0 {
        repo, err = w.repoService.GetRepositoryByID(ctx, uint(repoID))
    } else if repoName != "" {
        repo, err = w.repoService.GetRepositoryByName(ctx, repoName)
    } else {
        return mcp.NewToolResultError("需要提供 repo_id 或 repo_name"), nil
    }

    if err != nil {
        klog.Errorf("MCP: get repository failed: %v", err)
        return mcp.NewToolResultError(fmt.Sprintf("获取仓库失败: %v", err)), nil
    }

    // 获取文档列表
    docs, err := w.docService.GetDocumentsByRepoID(ctx, repo.ID)
    if err != nil {
        klog.Errorf("MCP: get documents failed: %v", err)
        return mcp.NewToolResultError(fmt.Sprintf("获取文档列表失败: %v", err)), nil
    }

    result := map[string]interface{}{
        "id":          repo.ID,
        "name":        repo.Name,
        "description": repo.Description,
        "url":         repo.URL,
        "branch":      repo.CloneBranch,
        "commit":      repo.CloneCommit,
        "status":      repo.Status,
        "documents":   docs,
    }

    data, _ := json.MarshalIndent(result, "", "  ")
    return mcp.NewToolResultText(string(data)), nil
}

// handleSearchDocuments 搜索文档
func (w *MCPServerWrapper) handleSearchDocuments(ctx context.Context, request mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    klog.V(6).Info("MCP: handleSearchDocuments called")

    query, err := request.RequireString("query")
    if err != nil {
        return mcp.NewToolResultError("query 参数是必需的"), nil
    }

    repoID, _ := request.RequireInt("repo_id")
    var repoIDPtr *uint
    if repoID > 0 {
        id := uint(repoID)
        repoIDPtr = &id
    }

    results, err := w.docService.SearchDocuments(ctx, query, repoIDPtr)
    if err != nil {
        klog.Errorf("MCP: search documents failed: %v", err)
        return mcp.NewToolResultError(fmt.Sprintf("搜索文档失败: %v", err)), nil
    }

    result := map[string]interface{}{
        "results": results,
        "total":   len(results),
    }

    data, _ := json.MarshalIndent(result, "", "  ")
    return mcp.NewToolResultText(string(data)), nil
}

// handleReadDocument 读取文档内容
func (w *MCPServerWrapper) handleReadDocument(ctx context.Context, request mcp.CallToolRequest) (*mcp.CallToolResult, error) {
    klog.V(6).Info("MCP: handleReadDocument called")

    docID, err := request.RequireInt("doc_id")
    if err != nil {
        return mcp.NewToolResultError("doc_id 参数是必需的"), nil
    }

    doc, err := w.docService.GetDocumentByID(ctx, uint(docID))
    if err != nil {
        klog.Errorf("MCP: read document failed: %v", err)
        return mcp.NewToolResultError(fmt.Sprintf("获取文档失败: %v", err)), nil
    }

    // 获取仓库名称
    repo, _ := w.repoService.GetRepositoryByID(ctx, doc.RepositoryID)
    repoName := ""
    if repo != nil {
        repoName = repo.Name
    }

    result := map[string]interface{}{
        "id":        doc.ID,
        "repo_id":   doc.RepositoryID,
        "repo_name": repoName,
        "title":     doc.Title,
        "filename":  doc.Filename,
        "content":   doc.Content,
        "branch":    doc.CloneBranch,
        "commit":    doc.CloneCommitID,
        "version":   doc.Version,
    }

    data, _ := json.MarshalIndent(result, "", "  ")
    return mcp.NewToolResultText(string(data)), nil
}
```

## 5. 路由配置

使用 mcp-go 提供的 SSE handler 集成到 Gin：

```go
// internal/handler/router.go

import (
    "github.com/mark3labs/mcp-go/server"
    "openDeepWiki/internal/mcp"
)

func SetupRouter(...) *gin.Engine {
    // ... 现有路由 ...

    // 创建 MCP Server
    mcpWrapper := mcp.NewMCPServer(repoService, docService)
    mcpServer := mcpWrapper.GetServer()

    // MCP SSE 端点
    mcpGroup := r.Group("/mcp")
    mcpGroup.GET("/sse", func(c *gin.Context) {
        // 使用 mcp-go 的 SSE handler
        handler := server.NewSSEServer(mcpServer)
        handler.ServeHTTP(c.Writer, c.Request)
    })

    return r
}
```

## 6. Service 层扩展

需要在 DocumentService 中添加搜索方法：

```go
// internal/service/document_service.go

type DocumentSearchResult struct {
    DocID    uint   `json:"doc_id"`
    RepoID   uint   `json:"repo_id"`
    RepoName string `json:"repo_name"`
    Title    string `json:"title"`
    Filename string `json:"filename"`
    Snippet  string `json:"snippet"`
}

// SearchDocuments 搜索文档内容
func (s *DocumentService) SearchDocuments(ctx context.Context, query string, repoID *uint) ([]DocumentSearchResult, error) {
    db := s.db.WithContext(ctx).Model(&model.Document{}).
        Where("is_latest = ?", true)

    if repoID != nil {
        db = db.Where("repository_id = ?", *repoID)
    }

    // 关键词搜索（title, filename, content）
    searchPattern := "%" + query + "%"
    db = db.Where("title LIKE ? OR filename LIKE ? OR content LIKE ?",
        searchPattern, searchPattern, searchPattern)

    var docs []model.Document
    if err := db.Limit(20).Find(&docs).Error; err != nil {
        return nil, err
    }

    // 构建结果
    results := make([]DocumentSearchResult, 0, len(docs))
    for _, doc := range docs {
        // 获取仓库名称
        repo, _ := s.repoService.GetRepositoryByID(ctx, doc.RepositoryID)
        repoName := ""
        if repo != nil {
            repoName = repo.Name
        }

        // 生成上下文片段
        snippet := generateSnippet(doc.Content, query, 200)

        results = append(results, DocumentSearchResult{
            DocID:    doc.ID,
            RepoID:   doc.RepositoryID,
            RepoName: repoName,
            Title:    doc.Title,
            Filename: doc.Filename,
            Snippet:  snippet,
        })
    }

    return results, nil
}

// generateSnippet 生成搜索结果片段
func generateSnippet(content, query string, maxLen int) string {
    // 查找关键词位置
    idx := strings.Index(strings.ToLower(content), strings.ToLower(query))
    if idx == -1 {
        if len(content) > maxLen {
            return content[:maxLen] + "..."
        }
        return content
    }

    // 计算片段起始位置
    start := idx - 50
    if start < 0 {
        start = 0
    }
    end := idx + len(query) + 150
    if end > len(content) {
        end = len(content)
    }

    snippet := content[start:end]
    if start > 0 {
        snippet = "..." + snippet
    }
    if end < len(content) {
        snippet = snippet + "..."
    }

    return snippet
}
```
