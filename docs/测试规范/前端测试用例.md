### 1.4 自动刷新

#### 测试用例 1.4.1: 每5秒自动刷新
```typescript
it('should refresh repository list every 5 seconds', async () => {
  const mockRepos = [{ id: 1, name: 'test-repo', url: 'https://github.com/test/repo', status: 'ready' }];
  let callCount = 0;

  const server = setupServer(
    rest.get('/api/repositories', (req, res, ctx) => {
      callCount++;
      return res(ctx.json(mockRepos));
    })
  );

  render(<Home />);

  // 等待初始调用
  await waitFor(() => callCount === 1);

  // 等待第二次调用（应该在5秒内发生）
  await waitFor(() => callCount === 2, { timeout: 6000 });

  expect(callCount).toBeGreaterThanOrEqual(2);

  server.close();
}, 10000);
```
