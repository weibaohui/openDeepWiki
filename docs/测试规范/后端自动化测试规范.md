# openDeepWiki 后端自动化测试规范

## 文档概述

本文档定义了 openDeepWiki 项目的 **后端自动化测试规范**，包括测试原则、工具选型、测试框架设计、编码规范和 AI 生成测试指南。

---

## 一、测试原则

### 1.1 核心原则

| 原则 | 说明 |
|------|------|
| **可预测性** | 测试结果应当稳定、可重复，避免 flaky 测试 |
| **自文档化** | 测试用例应当清晰描述被测功能，便于 AI 理解和生成 |
| **模块化** | 测试应当按功能模块组织，便于 AI 增量编写 |
| **隔离性** | 每个测试应当独立运行，避免依赖其他测试 |
| **快速反馈** | 单元测试应当快速运行（< 100ms） |
| **可维护性** | 测试代码应当易于更新和扩展 |

### 1.2 测试金字塔

```
        /\
       /E2E\          10% - 关键用户流程、API 集成
      /------\
     /集成测试\       30% - API 交互、数据库操作
    /----------\
   /  单元测试  \     60% - 函数、业务逻辑、模型方法
  /--------------\
```

### 4.1 Makefile 测试目标

```makefile
# 测试相关
.PHONY: test test-unit test-integration test-e2e test-coverage test-coverage-html test-bench

test:
	@echo "Running all tests..."
	go test -v -race -short ./...

test-unit:
	@echo "Running unit tests..."
	go test -v -short -run=TestUnit ./...

test-integration:
	@echo "Running integration tests..."
	go test -v -run=TestIntegration ./tests/integration/...

test-e2e:
	@echo "Running e2e tests..."
	go test -v -run=TestE2E ./tests/e2e/...

test-coverage:
	@echo "Running tests with coverage..."
	go test -v -coverprofile=coverage.out ./...
	go tool cover -func=coverage.out

test-coverage-html:
	@echo "Generating HTML coverage report..."
	go test -v -coverprofile=coverage.out ./...
	go tool cover -html=coverage.out -o coverage.html

test-bench:
	@echo "Running benchmarks..."
	go test -bench=. -benchmem ./...
```

func TestIntegration_DocumentFlow(t *testing.T) {
	router, db, _ := setupTestApp(t)

	t.Run("完整的文档创建流程", func(t *testing.T) {
		ctx := context.Background()

		// 1. 创建仓库
		repo := &model.Repository{
			Name: "测试仓库",
			URL:  "https://github.com/test/repo",
			Branch: "main",
		}
		err := db.Create(ctx, repo).Error
		require.NoError(t, err)

		// 2. 创建任务
		task := &model.Task{
			RepositoryID: repo.ID,
			Type:         "generate_document",
			Status:       "completed",
		}
		err = db.Create(ctx, task).Error
		require.NoError(t, err)

		// 3. 创建文档请求
		reqBody := map[string]interface{}{
			"repository_id": repo.ID,
			"task_id":       task.ID,
			"title":         "测试文档",
			"filename":      "test.md",
			"content":       "# Test",
		}
		body, _ := json.Marshal(reqBody)

		req, _ := http.NewRequest(http.MethodPost, "/api/v1/documents", bytes.NewReader(body))
		req.Header.Set("Content-Type", "application/json")

		w := httptest.NewRecorder()
		router.ServeHTTP(w, req)

		// 验证响应
		assert.Equal(t, http.StatusCreated, w.Code)

		var response map[string]interface{}
		err = json.Unmarshal(w.Body.Bytes(), &response)
		require.NoError(t, err)
		assert.Equal(t, float64(1), response["id"])

		// 验证数据库中的数据
		var doc model.Document
		err = db.First(&doc).Error
		require.NoError(t, err)
		assert.Equal(t, "测试文档", doc.Title)
	})

	t.Run("文档查询和更新流程", func(t *testing.T) {
		ctx := context.Background()

		// 1. 创建初始文档
		doc := &model.Document{
			RepositoryID: 1,
			Title:        "原始标题",
			Filename:     "test.md",
			Content:      "原始内容",
		}
		err := db.Create(ctx, doc).Error
		require.NoError(t, err)

		// 2. 通过 API 获取
		req, _ := http.NewRequest(http.MethodGet, "/api/v1/documents/"+fmt.Sprint(doc.ID), nil)
		w := httptest.NewRecorder()
		router.ServeHTTP(w, req)

		assert.Equal(t, http.StatusOK, w.Code)

		var response map[string]interface{}
		json.Unmarshal(w.Body.Bytes(), &response)

		// 3. 通过 API 更新
		updateReq := map[string]interface{}{
			"title": "更新标题",
		}
		body, _ := json.Marshal(updateReq)
		req, _ = http.NewRequest(http.MethodPut, "/api/v1/documents/"+fmt.Sprint(doc.ID), bytes.NewReader(body))
		req.Header.Set("Content-Type", "application/json")

		w = httptest.NewRecorder()
		router.ServeHTTP(w, req)

		assert.Equal(t, http.StatusOK, w.Code)

		// 4. 验证更新结果
		var updatedDoc model.Document
		err = db.First(&updatedDoc, doc.ID).Error
		require.NoError(t, err)
		assert.Equal(t, "更新标题", updatedDoc.Title)
	})
}
